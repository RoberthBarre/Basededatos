CREATE TABLE SERVICIO(
Id_ser INT  NOT NULL  ,
Acti_ser VARCHAR(30) NOT NULL,
CONSTRAINT pk_servicio PRIMARY KEY (Id_ser)
);

CREATE TABLE CUARTEL(
Id_cu INT NOT NULL ,
Nom_cu VARCHAR(45) NOT NULL,
Ubic_cu VARCHAR(45) NOT NULL,
CONSTRAINT pk_cuartel PRIMARY KEY (Id_cu)
);

CREATE TABLE CUERPOEJERCITO(
Id_ce INT NOT NULL ,
Denom_ce VARCHAR(45) NOT NULL,
CONSTRAINT pk_cuerpoejercito PRIMARY KEY (Id_ce)
);
CREATE TABLE COMPAÑIA(
Id_comp INT NOT NULL ,
Activi_comp VARCHAR(45) NOT NULL,
CONSTRAINT pk_compañia PRIMARY KEY (Id_comp)
);
CREATE TABLE CUARTEL_COMPAÑIA(
Id_cc INT NOT NULL ,
Id_cu INT NOT NULL,
Id_comp INT NOT NULL,
CONSTRAINT pk_cuartelcompañia PRIMARY KEY (Id_cc),
CONSTRAINT CUARTEL_CUARTELCOMPAÑIA_FK FOREIGN KEY(Id_cu) REFERENCES CUARTEL(Id_cu),
CONSTRAINT COMPAÑIA_CUARTELCOMPAÑIA_FK FOREIGN KEY(Id_comp) REFERENCES COMPAÑIA(Id_comp)
);
CREATE TABLE SOLDADO(
Id_s INT NOT NULL,
Nom_s VARCHAR(45) NOT NULL,
Ape_s VARCHAR(45) NOT NULL,
Id_ce INT NOT NULL ,
Id_comp INT NOT NULL ,
Id_cu INT NOT NULL ,
CONSTRAINT pk_soldado PRIMARY KEY (Id_s),
CONSTRAINT CUERPOEJERCITO_SOLDADO_FK FOREIGN KEY(Id_ce) REFERENCES CUERPOEJERCITO(Id_ce),
CONSTRAINT COMPAÑIA_SOLDADO_FK FOREIGN KEY(Id_comp) REFERENCES COMPAÑIA(Id_comp),
CONSTRAINT CUARTEL_SOLDADO_FK FOREIGN KEY(Id_cu) REFERENCES CUARTEL(Id_cu) 
);
CREATE TABLE SOLDADO_SERVICIO(
Id_ss INT NOT NULL ,
Fecha DATE NOT NULL,
Id_ser INT NOT NULL,
Id_s INT NOT NULL,
CONSTRAINT pk_soldadoservicio PRIMARY KEY (Id_ss),
CONSTRAINT SERVICIO_SOLDADOSERVICIO_FK FOREIGN KEY(Id_ser) REFERENCES SERVICIO(Id_ser),
CONSTRAINT SOLDADO_SOLDADOSERVICIO_FK FOREIGN KEY(Id_s) REFERENCES SOLDADO(Id_s)
);
